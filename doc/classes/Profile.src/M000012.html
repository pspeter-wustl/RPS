    def global_record
      values = {:wins => 0, :draws => 0, :losses => 0}
      @db.execute("SELECT wins, draws, losses FROM turns 
        GROUP BY game_id").each do |row|
        values[:wins] += row['wins']
        values[:draws] += row['draws']
        values[:losses] += row['losses']
      end
      values
    end

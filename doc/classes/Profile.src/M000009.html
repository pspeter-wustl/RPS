    def user_record
      values = {:wins => 0, :draws => 0, :losses => 0}
      @db.execute("SELECT turns.wins, turns.draws, turns.losses FROM 
        users, games, turns WHERE users.id = games.user_id AND 
        games.id = turns.game_id AND users.id = ? GROUP BY turns.game_id", 
        [@user_id]).each do |row|
        values[:wins] += row[0]
        values[:draws] += row[1]
        values[:losses] += row[2]
      end
      values
    end
